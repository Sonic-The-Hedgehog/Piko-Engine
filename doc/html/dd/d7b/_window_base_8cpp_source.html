201  LRESULT CALLBACK WindowBase::staticWndProc(HWND hwnd, UINT msg, WPARAM wParam, LPARAM lParam) {
202 
203  //std::cout << "Dispatch message." << std::endl;
204 
205  // Try to get the right message handler for the window.
206  auto seek = mWindowRegistry.find(hwnd);
207  if(seek == mWindowRegistry.end()) {
208  return DefWindowProc(hwnd, msg, wParam, lParam);
209  }
210 
211  WindowBase* window = seek->second;
212  bool msgConsumed = window->messageHandler(msg, wParam, lParam);
213 
214  return msgConsumed ? NULL : DefWindowProc(hwnd, msg, wParam, lParam);
